alias g='git'
alias gs='git status | more'
alias ga='git add '
alias gb='git branch '
alias gc='git commit'
alias gcnv="git commit --no-verify"
alias gd='git diff'
alias go='git checkout '
alias gl='git log'
alias gh='git log --pretty=format:"%h %ad | %s%d [%an]" --graph --date=short'
